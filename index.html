firebase.initializeApp(firebaseConfig); const auth = firebase.auth(); const db = firebase.firestore(); /* ========================= 🔐 AUTH ========================= */ function register() { const email = emailInput.value; const password = passwordInput.value; auth.createUserWithEmailAndPassword(email, password); } function login() { const email = emailInput.value; const password = passwordInput.value; auth.signInWithEmailAndPassword(email, password); } function logout() { auth.signOut(); } auth.onAuthStateChanged(user => { if (user) { document.getElementById("auth").style.display = "none"; document.getElementById("app").style.display = "block"; loadPlants(); } else { document.getElementById("auth").style.display = "block"; document.getElementById("app").style.display = "none"; } }); /* ========================= 🌱 PLANTEN ========================= */ async function addPlant() { const name = document.getElementById("name").value; const days = parseInt(document.getElementById("days").value); const user = auth.currentUser; await db.collection("plants").add({ userId: user.uid, name, interval: days, lastWatered: Date.now() }); loadPlants(); } async function loadPlants() { const user = auth.currentUser; const snapshot = await db.collection("plants") .where("userId", "==", user.uid) .get(); const container = document.getElementById("plants"); container.innerHTML = ""; snapshot.forEach(doc => { const plant = doc.data(); const now = Date.now(); const daysPassed = (now - plant.lastWatered) / (1000*60*60*24); const daysLeft = Math.ceil(plant.interval - daysPassed); const div = document.createElement("div"); div.className = "plant"; div.innerHTML = ` ${plant.name}
${daysLeft <= 0 ? "⚠️ Water nodig!" : daysLeft + " dagen over"}
`; container.appendChild(div); }); } async function waterPlant(id) { await db.collection("plants").doc(id).update({ lastWatered: Date.now() }); loadPlants(); } async function deletePlant(id) { await db.collection("plants").doc(id).delete(); loadPlants(); } /* ========================= 🔔 NOTIFICATIES ========================= */ function askNotification() { Notification.requestPermission(); } setInterval(checkPlants, 60000); async function checkPlants() { const user = auth.currentUser; if (!user) return; const snapshot = await db.collection("plants") .where("userId", "==", user.uid) .get(); snapshot.forEach(doc => { const plant = doc.data(); const now = Date.now(); const daysPassed = (now - plant.lastWatered) / (1000*60*60*24); if (daysPassed >= plant.interval) { if (Notification.permission === "granted") { new Notification("🌱 Water geven!", { body: plant.name + " heeft water nodig" }); } } }); }
